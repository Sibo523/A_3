int n, m;
    cin >> n >> m;
    vector<pair<int, int>> edges(m);

    for (int i = 0; i < m; ++i) {
        cin >> edges[i].second >> edges[i].first;
    }

    vector<vector<int>> scc = kosaraju(n, edges);

    for (const auto& component : scc) {
        for (int v : component) {
            cout << v << " ";
        }
        cout << endl;
    }

    return 0;
